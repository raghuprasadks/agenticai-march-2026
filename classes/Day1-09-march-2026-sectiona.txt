import cohere

co = cohere.ClientV2(api_key="<YOUR_API_KEY>")

response = co.chat(
    messages=[
        {
            "role": "system",
            "content": "you are a helpful bot"
        },
        {
            "role": "user",
            "content": [
                {
                    "type": "text",
                    "text": "who are you"
                }
            ]
        },
        {
            "role": "assistant",
            "content": [
                {
                    "type": "text",
                    "text": "I'm Command, an AI assistant, ready to provide thorough and informative responses. I've been developed by Cohere to offer helpful support. How can I be of assistance today?"
                }
            ]
        }
    ],
    temperature=0.3,
    model="command-a-03-2025",
)

print(response)
